/*
 ============================================================
  SPIDER ROBOT — 4-Leg, 2DOF per leg
  Board  : Arduino Nano V3
  Driver : PCA9685 (I2C address 0x40)
  Sensor : HC-SR04 ultrasonic
  Servos : 8 × SG90

  CHANNEL MAP (PCA9685):
    Ch0 = Front-Left  mid-leg (shoulder)
    Ch1 = Front-Left  foot
    Ch2 = Front-Right mid-leg
    Ch3 = Front-Right foot
    Ch4 = Rear-Left   mid-leg
    Ch5 = Rear-Left   foot
    Ch6 = Rear-Right  mid-leg
    Ch7 = Rear-Right  foot

  ARDUINO PINS:
    A4 = SDA (PCA9685)
    A5 = SCL (PCA9685)
    D9 = HC-SR04 Trig
    D10= HC-SR04 Echo
 ============================================================
*/

#include <Wire.h>
#include <Adafruit_PWMServoDriver.h>

Adafruit_PWMServoDriver pwm = Adafruit_PWMServoDriver(0x40);

// ---- SG90 pulse width (us) ----
#define SERVO_MIN   150    // ~0°
#define SERVO_MAX   600    // ~180°
#define SERVO_MID   375    // ~90°

// ---- HC-SR04 pins ----
#define TRIG_PIN    9
#define ECHO_PIN    10

// ---- obstacle distance threshold (cm) ----
#define OBSTACLE_CM 20

// ---- gait timing (ms) — increase for slower movement ----
#define STEP_DELAY  60     // delay between each sub-step
#define PAUSE       120    // pause at end of each full step

// ---- neutral (rest) angles for each channel ----
// Adjust these values after physical assembly so all legs stand level
int neutralAngle[8] = {90, 90, 90, 90, 90, 90, 90, 90};

// ---- leg geometry angles ----
// Mid-leg (shoulder) swing range from neutral
#define SHOULDER_SWING  25   // degrees either side
// Foot lift angle (up) and push angle (down)
#define FOOT_LIFT       35
#define FOOT_PUSH       20

// ============================================================
//  HELPERS
// ============================================================

// Convert angle (0-180) to PCA9685 pulse tick
uint16_t angleToPulse(int angle) {
  return map(angle, 0, 180, SERVO_MIN, SERVO_MAX);
}

// Set one servo channel to a given angle
void setServo(uint8_t channel, int angle) {
  angle = constrain(angle, 0, 180);
  pwm.setPWM(channel, 0, angleToPulse(angle));
}

// Move a servo channel smoothly from current→target in small steps
void smoothMove(uint8_t channel, int fromAngle, int toAngle, int stepDelay) {
  int step = (toAngle > fromAngle) ? 1 : -1;
  for (int a = fromAngle; a != toAngle; a += step) {
    setServo(channel, a);
    delay(stepDelay);
  }
  setServo(channel, toAngle);
}

// Stand all legs at neutral position
void standStill() {
  for (uint8_t ch = 0; ch < 8; ch++) {
    setServo(ch, neutralAngle[ch]);
  }
}

// Measure distance with HC-SR04 (returns cm)
long getDistance() {
  digitalWrite(TRIG_PIN, LOW);
  delayMicroseconds(2);
  digitalWrite(TRIG_PIN, HIGH);
  delayMicroseconds(10);
  digitalWrite(TRIG_PIN, LOW);
  long duration = pulseIn(ECHO_PIN, HIGH, 30000); // 30ms timeout
  if (duration == 0) return 999;                   // no echo = open space
  return duration * 0.034 / 2;
}

// ============================================================
//  GAIT — slow alternating diagonal (wave gait)
//
//  Diagonal pairs:
//    Group A: Front-Left (ch0,ch1)  + Rear-Right (ch6,ch7)
//    Group B: Front-Right (ch2,ch3) + Rear-Left  (ch4,ch5)
//
//  One step sequence:
//   1. Lift Group A feet → swing shoulders forward → lower feet
//   2. Push Group A shoulders back (body moves forward)
//   3. Repeat for Group B
// ============================================================

// Channel indices for each group
// {shoulder, foot}
const uint8_t legA[2][2] = {{0,1},{6,7}};  // FL, RR
const uint8_t legB[2][2] = {{2,3},{4,5}};  // FR, RL

// Forward swing offset direction per leg (mirror for left/right)
// +1 = positive direction of shoulder, -1 = negative
const int legAswingDir[2] = {1, -1};  // FL swings +, RR swings -
const int legBswingDir[2] = {-1, 1};  // FR swings -, RL swings +

void stepGroup(const uint8_t group[2][2], const int swingDir[2]) {
  // Phase 1: lift feet
  for (int i = 0; i < 2; i++) {
    uint8_t foot = group[i][1];
    smoothMove(foot,
               neutralAngle[foot],
               neutralAngle[foot] - FOOT_LIFT,
               STEP_DELAY / 3);
  }
  delay(20);

  // Phase 2: swing shoulders forward
  for (int i = 0; i < 2; i++) {
    uint8_t sh = group[i][0];
    smoothMove(sh,
               neutralAngle[sh],
               neutralAngle[sh] + swingDir[i] * SHOULDER_SWING,
               STEP_DELAY);
  }

  // Phase 3: lower feet
  for (int i = 0; i < 2; i++) {
    uint8_t foot = group[i][1];
    smoothMove(foot,
               neutralAngle[foot] - FOOT_LIFT,
               neutralAngle[foot] + FOOT_PUSH,
               STEP_DELAY / 3);
  }
  delay(20);

  // Phase 4: push back (propulsion) — other group plants and pushes
  for (int i = 0; i < 2; i++) {
    uint8_t sh = group[i][0];
    smoothMove(sh,
               neutralAngle[sh] + swingDir[i] * SHOULDER_SWING,
               neutralAngle[sh],
               STEP_DELAY);
  }

  // Phase 5: return feet to neutral
  for (int i = 0; i < 2; i++) {
    uint8_t foot = group[i][1];
    smoothMove(foot,
               neutralAngle[foot] + FOOT_PUSH,
               neutralAngle[foot],
               STEP_DELAY / 3);
  }
}

void walkForward() {
  stepGroup(legA, legAswingDir);
  delay(PAUSE);
  stepGroup(legB, legBswingDir);
  delay(PAUSE);
}

// ============================================================
//  TURN — swing both legs on one side forward, other backward
// ============================================================
void turnRight(int steps) {
  // Right turn: FL+RL swing forward, FR+RR swing backward
  const uint8_t right[2][2] = {{0,1},{4,5}}; // FL, RL
  const int rDir[2] = {1, 1};
  const uint8_t left[2][2] = {{2,3},{6,7}};  // FR, RR
  const int lDir[2] = {-1, -1};
  for (int s = 0; s < steps; s++) {
    stepGroup(right, rDir);
    stepGroup(left, lDir);
    delay(PAUSE);
  }
}

void turnLeft(int steps) {
  const uint8_t left[2][2] = {{2,3},{6,7}};
  const int lDir[2] = {1, 1};
  const uint8_t right[2][2] = {{0,1},{4,5}};
  const int rDir[2] = {-1, -1};
  for (int s = 0; s < steps; s++) {
    stepGroup(left, lDir);
    stepGroup(right, rDir);
    delay(PAUSE);
  }
}

// ============================================================
//  SETUP & LOOP
// ============================================================
void setup() {
  Serial.begin(9600);
  Serial.println("Spider Robot Starting...");

  pinMode(TRIG_PIN, OUTPUT);
  pinMode(ECHO_PIN, INPUT);

  pwm.begin();
  pwm.setOscillatorFrequency(27000000);
  pwm.setPWMFreq(50);  // SG90 runs at 50Hz

  delay(500);
  standStill();
  delay(1000);
  Serial.println("Ready.");
}

void loop() {
  long dist = getDistance();
  Serial.print("Distance: ");
  Serial.print(dist);
  Serial.println(" cm");

  if (dist > OBSTACLE_CM) {
    // Clear path — walk forward one gait cycle
    walkForward();
  } else {
    // Obstacle detected — stop, back off, turn
    Serial.println("Obstacle! Turning...");
    standStill();
    delay(300);
    turnRight(3);  // turn right 3 gait-steps
  }
}
